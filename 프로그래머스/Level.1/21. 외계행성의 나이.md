### [ 문제 풀이 ]
```JavaScript
function solution(age) {
    let alpha =['a','b','c','d','e','f','g','h','i','j'];
    let answer = '';
    // age는 정수이기 때문에 toString으로 문자화해준다.
    age = age.toString();
    for(let i = 0; i < age.length; i++){
        answer += alpha[age[i]]
    }
    return answer;
}
```

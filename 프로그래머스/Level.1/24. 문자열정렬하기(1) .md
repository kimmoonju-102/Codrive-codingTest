### [ 문제 풀이 ]
```JavaScript
function solution(my_string) {
    const stringArr = my_string.split('')
    const nums = stringArr.filter(ch => !isNaN(ch))

    const intArr = nums.map(num => Number(num));
    
    intArr.sort((a, b) => a - b)
     return intArr;
}
```
